-- Usar o banco de dados
USE luismi71_lendarios_bd;

-- Criar a tabela 'payments'
CREATE TABLE payments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    player_name VARCHAR(100) NOT NULL UNIQUE,
    type ENUM('Goleiro', 'Sim', 'Não') NOT NULL,
    status ENUM('OK', 'Pendente', 'Isento') NOT NULL DEFAULT 'Pendente',
    payment_date DATETIME DEFAULT NULL,
    month VARCHAR(7) DEFAULT NULL
);

-- Criar a tabela 'games'
CREATE TABLE games (
    id INT AUTO_INCREMENT PRIMARY KEY,
    location VARCHAR(100) NOT NULL,
    game_date VARCHAR(50) NOT NULL,
    game_time VARCHAR(10) NOT NULL
);
